SELECT b.species, COUNT(s.id) AS observations, AVG(b.weight) AS avg_weight
FROM birds b
LEFT JOIN sightings s ON b.id = s.bird_id
WHERE b.region = 'Europe'
GROUP BY b.species  
HAVING COUNT(s.id) > 3
ORDER BY observations DESC;
